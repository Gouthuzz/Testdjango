pipeline {
  agent any
  stages {
    stage('Checkout') {
      steps {
        deleteDir()
        checkout scm
      }
    }

    stage('Build Docker Image') {
      steps {
        sh 'docker build -t my-django-app:latest .'  // Replace with your desired image name and tag
      }
    }

    stage('Run Migrations') {
      steps {
        sh 'docker run --rm my-django-app:latest python manage.py migrate'
      }
    }

    stage('Run Tests') {
      steps {
        sh 'docker run --rm my-django-app:latest python manage.py test'
      }
    }

    stage('SonarQube Analysis') {
      // No changes needed here
    }

  }
  post {
    always {
      echo 'Pipeline completed'
    }
  }
}
